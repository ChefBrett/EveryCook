CREATE TABLE `recipes_voting` (
`RVO_ID` INT NOT NULL AUTO_INCREMENT PRIMARY KEY ,
`PRF_UID` INT NULL ,
`MEA_ID` INT NOT NULL ,
`COU_ID` INT NOT NULL ,
`REC_ID` INT NOT NULL ,
`RVO_COOK_DATE` INT NOT NULL ,
`RVO_VALUE` INT NULL ,
`RVR_ID` INT NOT NULL ,
`RVO_REASON` VARCHAR( 200 ) NULL
) ENGINE = InnoDB;


CREATE TABLE `recipes_voting_reason` (
`RVR_ID` INT NOT NULL AUTO_INCREMENT PRIMARY KEY ,
`RVR_DESC_EN_GB` VARCHAR( 200 ) NOT NULL ,
`RVR_DESC_DE_CH` VARCHAR( 200 ) NOT NULL
) ENGINE = InnoDB;
