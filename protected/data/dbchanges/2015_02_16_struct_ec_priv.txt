DELIMITER $$
CREATE TRIGGER ec_priv.`professional_insert` AFTER INSERT ON ec_priv.profiles
  FOR EACH ROW
  BEGIN
  IF NEW.PRF_ROLES LIKE '%professional%' THEN
    INSERT INTO ec.`professional_profiles` 
    SET 
      PRF_UID = NEW.PRF_UID, 
      PRF_FIRSTNAME = NEW.PRF_FIRSTNAME, 
      PRF_LASTNAME = NEW.PRF_LASTNAME, 
      PRF_LANG = NEW.PRF_LANG, 
      PRF_IMG_FILENAME = NEW.PRF_IMG_FILENAME, 
      PRF_IMG_ETAG = NEW.PRF_IMG_ETAG, 
      PRF_WORK_TITLE = NEW.PRF_WORK_TITLE, 
      PRF_WORK_LOCATION = NEW.PRF_WORK_LOCATION, 
      PRF_CUT_IDS = NEW.PRF_CUT_IDS, 
      PRF_PHILOSOPHY = NEW.PRF_PHILOSOPHY, 
      PRF_EXPERIENCE = NEW.PRF_EXPERIENCE, 
      PRF_AWARDS = NEW.PRF_AWARDS, 
      CREATED_BY = NEW.CREATED_BY, 
      CREATED_ON = NEW.CREATED_ON, 
      CHANGED_BY = NEW.CHANGED_BY, 
      CHANGED_ON = NEW.CHANGED_ON;
  END IF;
END$$

CREATE TRIGGER ec_priv.`professional_update` AFTER UPDATE ON ec_priv.profiles
  FOR EACH ROW
  BEGIN
  IF NEW.PRF_ROLES LIKE '%professional%' THEN
  	DELETE FROM ec.`professional_profiles` WHERE PRF_UID = OLD.PRF_UID;
    INSERT INTO ec.`professional_profiles` 
    SET 
      PRF_UID = NEW.PRF_UID, 
      PRF_FIRSTNAME = NEW.PRF_FIRSTNAME, 
      PRF_LASTNAME = NEW.PRF_LASTNAME, 
      PRF_LANG = NEW.PRF_LANG, 
      PRF_IMG_FILENAME = NEW.PRF_IMG_FILENAME, 
      PRF_IMG_ETAG = NEW.PRF_IMG_ETAG, 
      PRF_WORK_TITLE = NEW.PRF_WORK_TITLE, 
      PRF_WORK_LOCATION = NEW.PRF_WORK_LOCATION, 
      PRF_CUT_IDS = NEW.PRF_CUT_IDS, 
      PRF_PHILOSOPHY = NEW.PRF_PHILOSOPHY, 
      PRF_EXPERIENCE = NEW.PRF_EXPERIENCE, 
      PRF_AWARDS = NEW.PRF_AWARDS, 
      CREATED_BY = NEW.CREATED_BY, 
      CREATED_ON = NEW.CREATED_ON, 
      CHANGED_BY = NEW.CHANGED_BY, 
      CHANGED_ON = NEW.CHANGED_ON;
  ELSEIF OLD.PRF_ROLES LIKE '%professional%' THEN
  	DELETE FROM ec.`professional_profiles` WHERE PRF_UID = OLD.PRF_UID;
  END IF;
END$$

CREATE TRIGGER ec_priv.`professional_delete` AFTER DELETE ON ec_priv.profiles
  FOR EACH ROW
  BEGIN
  IF OLD.PRF_ROLES LIKE '%professional%' THEN
  	DELETE FROM ec.`professional_profiles` WHERE PRF_UID = OLD.PRF_UID;
  END IF;
END$$

DELIMITER ;
